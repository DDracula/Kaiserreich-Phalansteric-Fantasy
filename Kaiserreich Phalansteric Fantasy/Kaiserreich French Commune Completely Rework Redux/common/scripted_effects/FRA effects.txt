FRA_infulence_check = {
	if = {
		limit = {
			FROM = {
				capital_scope = {
					is_on_continent = europe
				}
			}
		}
		custom_effect_tooltip = infulence_change_eu
		add_to_variable = {
			FRA_infulence_europe = var:FRA_infulence_change
		}
	}
	else_if = {
		limit = {
			FROM = {
				capital_scope = {
					OR = {
						is_on_continent = asia
						is_on_continent = australia
						is_on_continent = india
					}
				}
			}
		}
		custom_effect_tooltip = infulence_change_as
		add_to_variable = {
			FRA_infulence_asia = var:FRA_infulence_change
		}
	}
	else_if = {
		limit = {
			FROM = {
				OR = {
					capital_scope = {
						is_on_continent = middle_east
					}
					original_tag = EGY
				}
			}
		}
		custom_effect_tooltip = infulence_change_me
		add_to_variable = {
			FRA_infulence_mideast = var:FRA_infulence_change
		}
	}
	else_if = {
		limit = {
			FROM = {
				capital_scope = {
					OR = {
						is_on_continent = north_america
						is_on_continent = central_america
						is_on_continent = south_america
					}
				}
			}
		}
		custom_effect_tooltip = infulence_change_am
		add_to_variable = {
			FRA_infulence_america = var:FRA_infulence_change
		}
	}
	else_if = {
		limit = {
			FROM = {
				capital_scope = {
					is_on_continent = africa
				}
				NOT = {
					original_tag = EGY
				}
			}
		}
		custom_effect_tooltip = infulence_change_af
		add_to_variable = {
			FRA_infulence_africa = var:FRA_infulence_change
		}
	}
}
