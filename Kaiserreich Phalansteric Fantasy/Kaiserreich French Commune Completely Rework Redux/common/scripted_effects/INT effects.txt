EUP_vote = {
	if = {
		limit = { has_country_flag = EUP_focus_union }
		set_temp_variable = { compare_value = eup_union }
	}
	if = {
		limit = { has_country_flag = EUP_focus_econ }
		set_temp_variable = { compare_value = eup_eco_receive }
	}
	if = {
		limit = { has_country_flag = EUP_focus_tech }
		set_temp_variable = { compare_value = eup_tech_receive }
	}
	if = {
		limit = { has_country_flag = EUP_focus_mili }
		set_temp_variable = { compare_value = eup_mili_receive }
	}
	set_variable = { EUP_discuss_agree = 0 }
	set_variable = { EUP_discuss_disagree = 0 }
	for_each_loop = {
		array = global.europe_member
		index = ti
		set_temp_variable = { tpossible = 50 }
		if = {
			limit = { tag = FRA }
			add_to_variable = { FRA.EUP_discuss_agree = global.europe_member_econ^ti }
		}
		else = {
			if = {
				limit = { FRA = { has_government = THIS } }
				add_to_temp_variable = { tpossible = 20 }
			}
			multiply_temp_variable = { FRA.compare_value = -0.8 }
			add_to_temp_variable = { FRA.compare_value = 48 }
			add_to_temp_variable = { tpossible = FRA.compare_value }
			random_list = {
				tpossible = {
					add_to_variable = { FRA.EUP_discuss_agree = global.europe_member_econ^ti }
				}
				100 = {
					modifier = {
						set_temp_variable = { t = tpossible }
						multiply_temp_variable = { t = -1 }
						add = t
					}
					add_to_variable = { FRA.EUP_discuss_disagree = global.europe_member_econ^ti }
				}
			}
		}
	}
}
