country_event = {
	id = comfrance.1
	title = comfrance.1.t
	desc = comfrance.1.desc
	picture = GFX_red_flag

	is_triggered_only = yes

	trigger = {
		OR = {
			is_ai = no
			has_game_rule = { rule = FRA_flag option = DEFAULT }
		}
	}

	option = {
		log = "[GetLogInfo]: event comfrance.1 option comfrance.1.a"
		name = comfrance.1.a
		ai_chance = {
			base = 1
			modifier = {
				factor = 0
				NOT = { has_country_leader = { character = FRA_benoit_frachon } }
			}
		}
		hidden_effect = {
			set_cosmetic_tag = FRA_alternate
			FRA_conseil_de_defense_federal = {
				set_portraits = {
					civilian = { large = GFX_portrait_FRA_conseil_de_defense_federal_civilian_large_alternate }
				}
			}
		}
	}

	option = {
		log = "[GetLogInfo]: event comfrance.1 option comfrance.1.b"
		name = comfrance.1.b
		ai_chance = {
			base = 1
			modifier = {
				factor = 0
				NOT = { has_country_leader = { character = FRA_pierre_monatte } }
				NOT = { has_country_leader = { character = FRA_conseil_de_defense_federal } }
			}
		}
		hidden_effect = {
			set_cosmetic_tag = FRA_red
		}
	}

	option = {
		log = "[GetLogInfo]: event comfrance.1 option comfrance.1.c"
		name = comfrance.1.c
		ai_chance = {
			base = 1
			modifier = {
				factor = 0
				NOT = { has_country_leader = { character = FRA_leon_jouhaux } }
				NOT = { has_country_leader = { character = FRA_georges_valois } }
			}
		}
		hidden_effect = {
			if = {
				limit = {
					OR = {
						has_cosmetic_tag = FRA_alternate
						has_cosmetic_tag = FRA_red
					}
				}
				drop_cosmetic_tag = yes
			}
			FRA_conseil_de_defense_federal = {
				set_portraits = {
					civilian = { large = GFX_portrait_FRA_conseil_de_defense_federal_civilian_large }
				}
			}
		}
	}
}
